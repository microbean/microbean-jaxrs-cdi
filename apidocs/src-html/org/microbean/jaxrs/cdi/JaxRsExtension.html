057public class JaxRsExtension implements Extension {
058
059  private final Set<Class<?>> potentialResourceClasses;
060
061  private final Set<Class<?>> potentialProviderClasses;
062
063  private final Set<Bean<?>> applicationBeans;
064  
065  private final Map<Class<?>, Bean<?>> resourceBeans;
066
067  private final Map<Class<?>, Bean<?>> providerBeans;
068
069  private final Set<Set<Annotation>> qualifiers;
070  
071  public JaxRsExtension() {
072    super();
073    this.potentialResourceClasses = new HashSet<>();
074    this.potentialProviderClasses = new HashSet<>();
075    this.applicationBeans = new HashSet<>();
076    this.resourceBeans = new HashMap<>();
077    this.providerBeans = new HashMap<>();
078    this.qualifiers = new HashSet<>();
079  }
080
081  private final <T> void discoverResourceClasses(@Observes
082                                                 @WithAnnotations({ Path.class })
083                                                 final ProcessAnnotatedType<T> event) {
084    if (event != null) {
085      final AnnotatedType<T> annotatedType = event.getAnnotatedType();
086      if (annotatedType != null) {
087        final Class<T> javaClass = annotatedType.getJavaClass();
088        if (javaClass != null) {
089          this.potentialResourceClasses.add(javaClass);
090        }
091      }
092    }
093  }
094
095  private final <T> void discoverProviderClasses(@Observes
096                                                 @WithAnnotations({ javax.ws.rs.ext.Provider.class })
097                                                 final ProcessAnnotatedType<T> event) {
098    if (event != null) {
099      final AnnotatedType<T> annotatedType = event.getAnnotatedType();
100      if (annotatedType != null) {
101        final Class<T> javaClass = annotatedType.getJavaClass();
102        if (javaClass != null) {
103          this.potentialProviderClasses.add(javaClass);
104        }
105      }
106    }
107  }
108
109  private final <T> void forAllEnabledBeans(@Observes
110                                            final ProcessBean<T> event) {
111    if (event != null) {
112      final Bean<T> bean = event.getBean();
113      if (bean != null) {
114        final Set<Type> beanTypes = bean.getTypes();
115        if (beanTypes != null && !beanTypes.isEmpty()) {
116          for (final Type beanType : beanTypes) {
117            final Class<?> beanTypeClass;
118            if (beanType instanceof Class) {
119              beanTypeClass = (Class<?>)beanType;
120            } else if (beanType instanceof ParameterizedType) {
121              final ParameterizedType parameterizedBeanType = (ParameterizedType)beanType;
122              final Type rawBeanType = parameterizedBeanType.getRawType();
123              if (rawBeanType instanceof Class) {
124                beanTypeClass = (Class<?>) rawBeanType;
125              } else {
126                beanTypeClass = null;
127              }
128            } else {
129              beanTypeClass = null;
130            }
131            if (beanTypeClass != null) {
132              if (Application.class.isAssignableFrom(beanTypeClass)) {
133                this.applicationBeans.add(bean);
134                this.qualifiers.add(bean.getQualifiers()); // yes, add the set as an element, not the set's elements
135              }
136
137              // Edge case: it could be an application whose methods
138              // are annotated with @Path, so it could still be a
139              // resource class.  That's why this isn't an else if.
140              if (this.potentialResourceClasses.remove(beanTypeClass)) {
141                // This bean has a beanType that we previously identified as a JAX-RS resource.
142                this.resourceBeans.put(beanTypeClass, bean);
143              }
144
145              if (this.potentialProviderClasses.remove(beanTypeClass)) {
146                // This bean has a beanType that we previously identified as a Provider class.
147                this.providerBeans.put(beanTypeClass, bean);
148              }
149            }
150          }
151        }
152      }
153    }
154  }
155
156  public final Set<Set<Annotation>> getApplicationQualifiers() {
157    return Collections.unmodifiableSet(this.qualifiers);
158  }
159
160  private final void afterNonSyntheticBeansAreEnabled(@Observes
161                                                      final AfterBeanDiscovery event,
162                                                      final BeanManager beanManager) {
163    if (event != null && beanManager != null) {
164      for (final Bean<?> bean : this.applicationBeans) {
165        assert bean != null;
166        @SuppressWarnings("unchecked")
167        final Bean<Application> applicationBean = (Bean<Application>)bean;
168        final CreationalContext<Application> cc = beanManager.createCreationalContext(applicationBean);
169        final Class<? extends Annotation> applicationScope = applicationBean.getScope();
170        assert applicationScope != null;
171        Context context = beanManager.getContext(applicationScope);        
172        assert context != null;
173        AlterableContext alterableContext = context instanceof AlterableContext ? (AlterableContext)context : null;
174        Application application = null;                
175        try {
176          if (alterableContext == null) {
177            application = applicationBean.create(cc);
178          } else {
179            try {
180              application = alterableContext.get(applicationBean, cc);
181            } catch (final ContextNotActiveException ok) {
182              alterableContext = null;
183              application = applicationBean.create(cc);
184            }
185          }
186          if (application != null) {            
187            final Set<Class<?>> classes = application.getClasses();
188            if (classes != null && !classes.isEmpty()) {
189              final Set<Annotation> applicationQualifiers = applicationBean.getQualifiers();
190              for (final Class<?> cls : classes) {
191                final Object resourceBean = this.resourceBeans.remove(cls);
192                final Object providerBean = this.providerBeans.remove(cls);
193                if (resourceBean == null && providerBean == null) {
194                  event.addBean()
195                    .scope(Dependent.class) // by default; possibly overridden by read()
196                    .read(beanManager.createAnnotatedType(cls))
197                    .addQualifiers(applicationQualifiers);
198                }
199              }
200            }
201            // Deliberately don't try to deal with getSingletons().
202          }
203        } finally {
204          try {
205            if (application != null) {
206              if (alterableContext == null) {
207                applicationBean.destroy(application, cc);
208              } else {
209                try {
210                  alterableContext.destroy(applicationBean);
211                } catch (final UnsupportedOperationException ok) {
212
213                }
214              }
215            }
216          } finally {
217            cc.release();
218          }
219        }
220      }
221      this.applicationBeans.clear();
222
223      // Any potentialResourceClasses left over here are annotated
224      // types we discovered, but for whatever reason were not made
225      // into beans.  Maybe they were vetoed.
226      this.potentialResourceClasses.clear();
227
228      // Any potentialProviderClasses left over here are annotated
229      // types we discovered, but for whatever reason were not made
230      // into beans.  Maybe they were vetoed.
231      this.potentialProviderClasses.clear();
232      
233      // OK, when we get here, if there are any resource beans left
234      // lying around they went "unclaimed".  Build a synthetic
235      // Application for them.
236      if (!this.resourceBeans.isEmpty()) {
237        final Set<Entry<Class<?>, Bean<?>>> resourceBeansEntrySet = this.resourceBeans.entrySet();
238        assert resourceBeansEntrySet != null;
239        assert !resourceBeansEntrySet.isEmpty();
240        final Map<Set<Annotation>, Set<Class<?>>> resourceClassesByQualifiers = new HashMap<>();
241        for (final Entry<Class<?>, Bean<?>> entry : resourceBeansEntrySet) {
242          assert entry != null;
243          final Set<Annotation> qualifiers = entry.getValue().getQualifiers();
244          Set<Class<?>> resourceClasses = resourceClassesByQualifiers.get(qualifiers);
245          if (resourceClasses == null) {
246            resourceClasses = new HashSet<>();
247            resourceClassesByQualifiers.put(qualifiers, resourceClasses);
248          }
249          resourceClasses.add(entry.getKey());
250        }
251
252        final Set<Entry<Set<Annotation>, Set<Class<?>>>> entrySet = resourceClassesByQualifiers.entrySet();
253        assert entrySet != null;
254        assert !entrySet.isEmpty();
255        for (final Entry<Set<Annotation>, Set<Class<?>>> entry : entrySet) {
256          assert entry != null;
257          final Set<Annotation> resourceBeanQualifiers = entry.getKey();
258          final Set<Class<?>> resourceClasses = entry.getValue();
259          assert resourceClasses != null;
260          assert !resourceClasses.isEmpty();
261          final Set<Class<?>> allClasses;
262          if (this.providerBeans.isEmpty()) {
263            allClasses = resourceClasses;
264          } else {
265            allClasses = new HashSet<>(resourceClasses);
266            final Set<Entry<Class<?>, Bean<?>>> providerBeansEntrySet = this.providerBeans.entrySet();
267            assert providerBeansEntrySet != null;
268            assert !providerBeansEntrySet.isEmpty();
269            final Iterator<Entry<Class<?>, Bean<?>>> providerBeansIterator = providerBeansEntrySet.iterator();
270            assert providerBeansIterator != null;
271            while (providerBeansIterator.hasNext()) {
272              final Entry<Class<?>, Bean<?>> providerBeansEntry = providerBeansIterator.next();
273              assert providerBeansEntry != null;
274              final Set<Annotation> providerBeanQualifiers = providerBeansEntry.getValue().getQualifiers();
275              boolean match = false;
276              if (resourceBeanQualifiers == null) {
277                if (providerBeanQualifiers == null) {
278                  match = true;
279                }
280              } else if (resourceBeanQualifiers.equals(providerBeanQualifiers)) {
281                match = true;
282              }
283              if (match) {
284                allClasses.add(providerBeansEntry.getKey());
285                providerBeansIterator.remove();
286              }              
287            }
288          }
289
290          event.addBean()
291            .addTransitiveTypeClosure(SyntheticApplication.class)
292            .scope(Singleton.class)
293            .addQualifiers(resourceBeanQualifiers)
294            .createWith(cc -> new SyntheticApplication(allClasses));
295          this.qualifiers.add(resourceBeanQualifiers);
296        }
297
298        this.resourceBeans.clear();
299      }
300
301      if (!this.providerBeans.isEmpty()) {
302        // TODO: we found some provider class beans but never
303        // associated them with any application.  This would only
304        // happen if they were not qualified with qualifiers that also
305        // qualified unclaimed resource beans.  That would be odd.
306        // Either we should throw a deployment error or just ignore
307        // them.
308      }
309      this.providerBeans.clear();
310
311    }
312  }
313
314  public static final class SyntheticApplication extends Application {
315
316    private final Set<Class<?>> classes;
317    
318    SyntheticApplication(final Set<Class<?>> classes) {
319      super();
320      this.classes = classes;
321    }
322
323    @Override
324    public final Set<Class<?>> getClasses() {
325      return this.classes;
326    }
327    
328  }
