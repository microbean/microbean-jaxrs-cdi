041@ApplicationScoped
042public class TestApplicationDiscovery {
043
044  private AutoCloseable container;
045  
046  public TestApplicationDiscovery() {
047    super();
048  }
049
050  @Before
051  public void startContainer() throws Exception {
052    this.stopContainer();
053    final SeContainerInitializer initializer = SeContainerInitializer.newInstance();
054    initializer.disableDiscovery();
055    initializer.addExtensions(JaxRsExtension.class);
056    initializer.addBeanClasses(this.getClass(), MyApplication.class);
057
058    // Add a bean class for a resource class that is already "claimed"
059    // by MyApplication.
060    initializer.addBeanClasses(MyResource.class);
061
062    // Add a "free-floating" resource classe.
063    initializer.addBeanClasses(UnclaimedResource.class);
064    
065    this.container = initializer.initialize();
066  }
067
068  @After
069  public void stopContainer() throws Exception {
070    if (this.container != null) {
071      this.container.close();
072      this.container = null;
073    }
074  }
075
076  private static final void onStartup(@Observes @Initialized(ApplicationScoped.class) final Object event,
077                                      final MyApplication myApplication,
078                                      final JaxRsExtension.SyntheticApplication syntheticApplication) {
079    assertNotNull(myApplication);
080    assertNotNull(syntheticApplication);
081  }
082

085
086  }

096    }

098    @Override

103  }
